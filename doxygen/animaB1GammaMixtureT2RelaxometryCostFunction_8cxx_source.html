142 {
143  unsigned int numT2Signals = m_T2RelaxometrySignals.size();
144  unsigned int numDistributions = m_GammaMeans.size();
145 
146  m_SigmaSquare = 0.0;
147 
148  for (unsigned int i = 0;i < numT2Signals;++i)
149  {
150  m_Residuals[i] = m_T2RelaxometrySignals[i];
151  for (unsigned int j = 0;j < numDistributions;++j)
152  m_Residuals[i] -= m_PredictedSignalAttenuations(i,j) * m_OptimalT2Weights[j];
